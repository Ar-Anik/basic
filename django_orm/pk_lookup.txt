pk = primary key lookup

# Below three statements are equivalent:
>>> Blog.objects.get(id__exact=1)
<Blog: Tech World>

>>> Blog.objects.get(id=1)          # __exact is implied
<Blog: Tech World>

>>> Blog.objects.get(pk=1)          # pk implies id__exact
<Blog: Tech World>


# We can combine any query term with pk
>>> Blog.objects.filter(pk__in=[1, 3])        # converted like id__in when id is primary key
<QuerySet [<Blog: Tech World>, <Blog: Wanderlust>]>

>>> Blog.objects.filter(pk__gt=1)             # converted like id__gt when id is primary key
<QuerySet [<Blog: Food Paradise>, <Blog: Wanderlust>]>


# pk lookups also work across joins. For example, these three statements are equivalent:
>>> Entry.objects.filter(blog__id__exact=3)
<QuerySet [<Entry: Wanderlust Travel Guide>]>

>>> Entry.objects.filter(blog__id=3)        # add __exact
<QuerySet [<Entry: Wanderlust Travel Guide>]>

>>> Entry.objects.filter(blog__pk=3)        # add __id__exact
<QuerySet [<Entry: Wanderlust Travel Guide>]>
